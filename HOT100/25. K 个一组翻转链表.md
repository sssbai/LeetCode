# [25. K 个一组翻转链表](https://leetcode.cn/problems/reverse-nodes-in-k-group/description/?envType=study-plan-v2&envId=top-100-liked)

给你链表的头节点 `head` ，每`k`个节点一组进行翻转，请你返回修改后的链表。

`k` 是一个正整数，它的值小于或等于链表的长度。如果节点总数不是`k`的整数倍，那么请将最后剩余的节点保持原有顺序。

你不能只是单纯的改变节点内部的值，而是需要实际进行节点交换。

**提示：**

- `1 <= k <= n <= 5000`
- `0 <= Node.val <= 1000`

**进阶：** 你可以设计一个只用 `O(1)` 额外内存空间的算法解决此问题吗？

## Solution

可以直接遍历统计多少个k，之后直接在while中进行翻转

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    pair<ListNode*, ListNode*> reverseList(ListNode* node){
        ListNode* dummy = new ListNode();
        dummy->next = node;
        ListNode* tail = node;
        while(node){
            ListNode* tmp = node;
            node = node->next;
            tmp->next = dummy->next;
            dummy->next = tmp;
        }
        return {dummy->next, tail};
    }
    ListNode* getk(ListNode* node, int k){
        ListNode* dummy = new ListNode();
        dummy->next = node;
        ListNode* pre = dummy;
        while(k--){
            if(!node)return dummy->next;
            pre = pre->next;
            node = node->next;
        }
        pre->next = nullptr;
        return node;
        
    }
    ListNode* reverseKGroup(ListNode* head, int k) {
        ListNode* dummy = new ListNode();
        dummy->next = head;
        ListNode* cur = head;
        ListNode* pre = dummy;
        while(cur){
            ListNode* next = getk(cur, k);
            if(next == cur)break;
            auto [new_cur, new_tail] = reverseList(cur);
            pre->next = new_cur;
            new_tail->next = next;
            cur = next;
            pre = new_tail;
        }
        return dummy->next;
    }
};
```

### 最佳版本

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */

class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        // Step 1: 计算链表长度
        int length = 0;
        ListNode* node = head;
        while (node) {
            length++;
            node = node->next;
        }

        // Step 2: 创建 dummy 节点，方便操作头节点
        ListNode* dummyHead = new ListNode(0);
        dummyHead->next = head;

        ListNode* prevGroupTail = dummyHead;
        int numGroups = length / k;

        // Step 3: 每次反转 k 个节点
        while (numGroups--) {
            ListNode* groupHead = prevGroupTail->next;
            ListNode* nextNode = groupHead->next;

            // 使用头插法反转当前组的节点
            for (int i = 1; i < k; ++i) {//只有k-1次
                groupHead->next = nextNode->next;//将这个头结点指向第i个节点
              
                nextNode->next = prevGroupTail->next;//将第i个节点插入前面
                prevGroupTail->next = nextNode;
              
                nextNode = groupHead->next;//获取下一个节点
            }

            // 更新 prevGroupTail 指向当前组的尾部
            prevGroupTail = groupHead;
        }

        return dummyHead->next;
    }
};
```
